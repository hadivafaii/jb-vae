def make_grid(
		x: Union[np.ndarray, torch.Tensor],
		grid_size: Union[int, Tuple[int, int]],
		pad: int = 1,
		pad_val: float = np.nan,
		normalize: bool = True,
		**kwargs, ):
	x = to_np(x)
	if x.ndim == 4:
		x = x[:, 0, ...]
	_, h, w = x.shape

	if isinstance(grid_size, int):
		grid_size = (grid_size, grid_size)
	n_rows, n_cols = grid_size

	grid = np.ones((
		(h + pad) * n_rows - pad,
		(w + pad) * n_cols - pad,
	)) * pad_val

	for idx in range(min(n_rows * n_cols, len(x))):
		i = idx // n_cols
		j = idx % n_cols
		a = (h + pad) * i
		b = (w + pad) * j

		y = x[idx]
		if normalize:
			y = normalize_img(
				y, **kwargs)

		grid[a:a + h, b:b + w] = y

	return grid


def normalize_img(
		x: np.ndarray,
		method: str = 'min-max',
		val_range: Tuple[float, float] = (0, 1), ):
	if method == 'min-max':
		xmin = np.min(x)
		xmax = np.max(x)
		x_nrm = (x - xmin) / (xmax - xmin)

		a, b = min(val_range), max(val_range)
		x_nrm = x_nrm * (b - a) + a

	elif method == 'abs-max':
		x_nrm = x / np.max(np.abs(x))
	else:
		raise ValueError(method)

	return x_nrm
