def _show_pairwise(i, j, key='r*dr'):
    df = pd.DataFrame({
        f'z-{i}': etc[key][:, i],
        f'z-{j}': etc[key][:, j],
        f'z-{i}_log': np.log(np.abs(etc[key][:, i])),
