for f in outside_files:
    outside_efficiency = dict(np.load(

        allow_pickle=True,
    ))
    mse = (vld - outside_efficiency['recons_val']) ** 2
    mse = mse.sum(axis=(2, 3))
    mu, sd = mse.mean(), mse.std()
    mu, sd = f"{mu:0.2f}", f"{sd:0.2f}".lstrip('0')
    entry_mse = f"\entry{'{'}{mu}{'}'}{'{'}{sd}{'}'}"
    
    entry_lifetime = 0.0

    posterior_params = torch.tensor(outside_efficiency['posteriors_val']).squeeze()
    posterior_mu, posterior_logvar = posterior_params[:, 0, :], posterior_params[:, 1, :]
    dist = Normal(
        loc=posterior_mu,
        log_scale=posterior_logvar.mul(0.5),
        clamp=None,
        temp=1.0,
    )
    z = dist.sample()

    lifetime = sparse_score(z.abs())[0]
    mu, sd = lifetime.mean(), lifetime.std()
    mu, sd = f"{mu:0.2f}", f"{sd:0.2f}".lstrip('0')
    entry_lifetime = f"\entry{'{'}{mu}{'}'}{'{'}{sd}{'}'}"

    eps = torch.finfo(torch.float32).eps
    percent_zero = z.abs() < eps
    percent_zero = percent_zero.float().mean().item() * 100

    s = '\n'.join([
        f"# params: {outside_efficiency['model_n_params'].item()/1e6:0.2f} M",
        f"MSE = {entry_mse}",
        f"lifetime = {entry_lifetime}",
        f"% zero = {percent_zero:0.1g}",
    ])
    alt_models_str[f] = s

    print(f"{f}\n\n{s}\n\n")
