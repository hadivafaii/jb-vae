def do_the_loop():
    for name in fits:
        tr, meta = load_quick(
            pjoin(add_home(root), name),
            device=device,
            verbose=False,
            lite=True,
        )
        data, loss, etc = tr.validate()
        dead = tr.find_dead_neurons(loss['kl_diag'])
    
        
        msg = f"{name} ——— total # dead: {dead.sum()}"
        print(msg)
        if tr.model.cfg.type == 'categorical':
            tr.model.show(dpi=70)
            continue
        
        w_dec = tonp(tr.model.fc_dec.weight)
        norms = sp_lin.norm(w_dec, axis=0)
        w_dec = w_dec.T.reshape(tr.model.shape).squeeze()
        
        kl = loss['kl_diag']
        order = np.argsort(kl)
        
        w_dec = w_dec[order]
        norms = norms[order]
        dead = dead[order]
        kl = kl[order]
        
        fig, ax = create_figure(1, 1, (4, 2.2))
        bins = np.linspace(0, max(kl), 121)
        sns.histplot(kl[~dead], bins=bins, color='dimgrey', label='alive', ax=ax)
        sns.histplot(kl[dead], bins=bins, color='r', alpha=0.8, label='dead', ax=ax)
        ax.locator_params(axis='x', nbins=11)
        ax.set(yscale='log')
        ax.set_title(f"total # dead: {dead.sum()}", fontsize=14)
        ax.ticklabel_format(
            axis='x',
            style='sci',
            scilimits=(0, 0),
        )
        ax.legend()
        plt.show()
    
        _ = tr.model.show(dpi=50, order=order)
        _ = plot_weights(w_dec, scaling=(~dead).astype(float), dpi=50)
        print('_' * 80)
        print('_' * 80)
        print('\n\n')
