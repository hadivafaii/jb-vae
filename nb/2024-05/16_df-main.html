def accept_fn(name):
    archi = name.split('-')[1]
    accept = (
        archi in ['<lin|lin>', '<conv+b|lin>']
        and '-b' not in name
    )
    return accept


def accept_fn_etc(name):
    archi = name.split('-')[1]
    accept = (
        archi in ['<lin|lin>', '<conv+b|lin>']
        and any(fn in name for fn in ['relu', 'exp'])
        and 'gaussian' in name
        and '-b' not in name
    )
    return accept
