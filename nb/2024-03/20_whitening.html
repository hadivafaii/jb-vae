log_line = np.log(kvals) * lin_reg.slope + lin_reg.intercept

plt.loglog(kvals, psd[good].mean(0), marker='o', lw=1.2, color='dimgrey')
plt.loglog(kvals, np.exp(log_line), color='r', lw=2, ls='-')
plt.grid()
plt.show()
